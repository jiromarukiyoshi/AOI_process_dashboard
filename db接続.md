
  CREATE OR REPLACE FORCE EDITIONABLE VIEW "OFCSEUC"."FCSEUC_TRACE_SUMMARY_V" ("FFC案件NO", "FFC案件名", "Fコード", "FFC出荷NO", "枝番", "Tコード", "顧客名", "店舗番号", "店名", "製品コード", "製品名(型番)", "シリアルNO", "シリアル取得有無", "15030勘定振替", "FFC出荷日", "設置日", "ステータス（FFC出荷No）", "ステータス（Fコード）", "FFC経由") AS 
  SELECT FFP.FFC_PROJECT_NO "FFC案件NO"
      ,FFP.FFC_PROJECT_NAME "FFC案件名"
      ,FST.FFC_SHIPMENT_NO || '-' || TRIM(TO_CHAR(FFC.FFC_EDA_NO, '000')) "Fコード"
      ,FST.FFC_SHIPMENT_NO "FFC出荷NO"
      ,TO_CHAR(FFC.FFC_EDA_NO,'000') "枝番"
      ,FFP.T_CODE "Tコード"
      ,ZCS.KOKYAKU_NAME "顧客名"
      ,FST.STORE_CODE "店舗番号"
      ,FST.STORE_NAME "店名"
      ,FIS.ITEM_CODE "製品コード"
      ,FIM.SHOHIN_NAME "製品名(型番)"
      ,FIS.SERIAL_NO "シリアルNO"
      ,FFV1.FIXED_VALUE "シリアル取得有無"
      ,FFV2.FIXED_VALUE "15030勘定振替"
      ,FST.FFC_SHIPMENT_DATE "FFC出荷日"
      ,FST.INSTALLATION_DATE "設置日"
      ,FFV3.FIXED_VALUE "ステータス（FFC出荷No）"
      ,FFV4.FIXED_VALUE "ステータス（Fコード）"
      ,FFV5.FIXED_VALUE "FFC経由"
  FROM OFCSEUC.FCS_FFC_PROJECT_V FFP
      ,OFCSEUC.FCS_SCHEDULE_TRN_V FST
      ,OFCSEUC.FCS_F_CODE_V FFC
      ,OFCSEUC.FCS_ITEM_SERIAL_V FIS
      ,OFCSEUC.ZUS_XX00_S_CUSTOMER_SUBS_ALL_V ZCS
      ,OFCSEUC.FCS_ITEM_MST_V FIM
     ,(SELECT KEY3 ,FIXED_VALUE
          FROM FCS.FCS_FIX_VALUES
         WHERE 1 = 1
           AND KEY1 = 'FCS_COMMON'
           AND KEY2 = 'IS_SERIAL_REQUIRED_KBN' ) FFV1
     ,(SELECT KEY3 ,FIXED_VALUE
          FROM FCS.FCS_FIX_VALUES
         WHERE 1 = 1
           AND KEY1 = 'FCS_COMMON'
           AND KEY2 = 'HAS_ACCT_TRANSFER_KBN' ) FFV2
      ,(SELECT KEY3 ,FIXED_VALUE
          FROM FCS.FCS_FIX_VALUES
         WHERE 1 = 1
           AND KEY1 = 'FCS_COMMON'
           AND KEY2 = 'SCHEDULE_STATUS' ) FFV3
      ,(SELECT KEY3 ,FIXED_VALUE
          FROM FCS.FCS_FIX_VALUES
         WHERE 1 = 1
           AND KEY1 = 'FCS_COMMON'
           AND KEY2 = 'F_CODE_STATUS' ) FFV4
      ,(SELECT KEY3 ,FIXED_VALUE
          FROM FCS.FCS_FIX_VALUES
         WHERE 1 = 1
           AND KEY1 = 'FCS_COMMON'
           AND KEY2 = 'VIA_FFC_KBN' ) FFV5
 WHERE 1 = 1
   AND FFP.FFC_PROJECT_NO = FST.FFC_PROJECT_NO
   AND FFP.T_CODE = ZCS.T_CODE
   AND FST.FFC_SHIPMENT_NO = FFC.FFC_SHIPMENT_NO
   AND FFC.FFC_SHIPMENT_NO = FIS.FFC_SHIPMENT_NO
   AND FFC.FFC_EDA_NO = FIS.FFC_EDA_NO
   AND FIS.ITEM_CODE = FIM.SHOHIN_CODE
   AND FFV1.KEY3 = FIS.IS_SERIAL_REQUIRED_KBN
-- AND FFV2.KEY3 = FST.HAS_ACCT_TRANSFER_KBN  --2026/03/30 DELETE
   AND FFV2.KEY3 = FIS.HAS_ACCT_TRANSFER_KBN  --2026/03/30 ADD
   AND FFV3.KEY3 = FST.SCHEDULE_STATUS
   AND FFV4.KEY3 = FFC.F_CODE_STATUS
   AND FFV5.KEY3 = FIS.VIA_FFC_KBN
   AND FIS.DELETE_FLAG = '0'
   --and FST.FFC_SHIPMENT_NO in ('F000100001','F000100002')
ORDER BY "Fコード" ,"製品コード"
WITH READ ONLY
;
